using namespace std; 

// Structure to hold book information
struct Book {
    string title;
    string summary;
};

// Structure to hold author information
struct Author {
    string name;
    vector<Book> books; // List of books by the author
};

// Structure to hold category information
struct Category {
    string name;
    vector<Author> authors; // List of authors in the category
};

// Function to display books of a selected author
void displayBooks(const Author& author) {
    cout << "Books by " << author.name << ":\n";
    for (const auto& book : author.books) {
        cout << "Title: " << book.title << "\nSummary: " << book.summary << "\n\n";
    }
}

// Function to display authors of a selected category
void displayAuthors(const Category& category) {
    cout << "Authors in " << category.name << " category:\n";
    for (size_t i = 0; i < category.authors.size(); ++i) {
        cout << i + 1 << ". " << category.authors[i].name << "\n";
    }
}

// Function to display categories
void displayCategories(const vector<Category>& categories) {
    cout << "Select a category:\n";
    for (size_t i = 0; i < categories.size(); ++i) {
        cout << i + 1 << ". " << categories[i].name << "\n";
    }
}
